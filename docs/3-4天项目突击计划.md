**解决方案**: 按功能分类组织文件

#### 💡 每个文件夹的作用详解

##### 1️⃣ `data/` - 数据文件夹
**为什么单独放数据?**
- 数据文件很大(10万条),和代码混在一起不合适
- 原始数据和清洗后数据分开,方便对比
- 可以随时替换数据集,不影响代码

**包含文件**:
- `ecommerce_orders.csv`: 原始数据,不要修改
- `ecommerce_orders_cleaned.csv`: 清洗后的数据,用于分析

**面试时可以说**: "我把原始数据和清洗后数据分开存储,这样可以在数据出现问题时快速对比,也方便后续用不同数据集进行分析。"

---

#### ⚠️ 常见错误

❌ **错误1**: 所有代码写在一个文件
```python
# 不要这样!所有代码都在main.py里
# 数据生成、清洗、分析、可视化全在一个文件
```

❌ **错误2**: 文件命名不清楚
```python
# 不要这样!
data.py      # 不知道这个文件做什么
code.py      # 太笼统
script.py    # 没有功能描述
```

✅ **正确做法**:
```python
# 按功能命名,清晰明了
01_generate_data.py    # 一看就知道是生成数据
02_clean_data.py       # 数据清洗
03_export_to_mysql.py  # 导出到MySQL
```

❌ **错误3**: 使用绝对路径
```python
# 不要这样!换台电脑就跑不起来了
df = pd.read_csv('D:/27860/面试练习/xxxx/data/ecommerce_orders.csv')
```

✅ **正确做法**:
```python
# 使用相对路径,项目可以随处运行
df = pd.read_csv('data/ecommerce_orders.csv')
```

```python
import pandas as pd
import numpy as np
from datetime import datetime, timedelta
import random

# 生成模拟电商订单数据

# 时间范围: 2024年全年

end_date = datetime(2024, 12, 31)

n_orders = 100000

orders_df = pd.DataFrame({
    'order_id': range(1, n_orders + 1),
    'user_id': np.random.randint(1, 5000, n_orders),
    'product_id': np.random.randint(1, 200, n_orders),
    'product_name': [f'商品_{i}' for i in np.random.randint(1, 200, n_orders)],
    'category': np.random.choice(['电子产品', '服装', '食品', '家居', '图书'], n_orders),
    'quantity': np.random.randint(1, 5, n_orders),
    'price': np.random.uniform(10, 500, n_orders).round(2),
    'discount_rate': np.random.choice([0, 0.05, 0.1, 0.15, 0.2], n_orders),
    'order_date': np.random.choice(date_range, n_orders),
    'city': np.random.choice(['北京', '上海', '广州', '深圳', '杭州', '成都', '武汉', '西安'], n_orders),
    'payment_method': np.random.choice(['支付宝', '微信', '银行卡'], n_orders)
})

# 计算订单金额
orders_df['total_amount'] = (orders_df['quantity'] * orders_df['price']).round(2)
orders_df['discount_amount'] = (orders_df['total_amount'] * orders_df['discount_rate']).round(2)
orders_df['final_amount'] = (orders_df['total_amount'] - orders_df['discount_amount']).round(2)

# 保存为CSV
orders_df.to_csv('ecommerce_orders.csv', index=False, encoding='utf-8-sig')

print(f"✅ 已生成 {n_orders} 条订单数据")

```

```python
import numpy as np

# 读取数据
df = pd.read_csv('ecommerce_orders.csv')

# 数据清洗步骤

# 1. 去除重复数据
df = df.drop_duplicates()
print(f"去重后数据量: {len(df)} 条")

print("\n缺失值统计:")
print(df.isnull().sum())

# 填充缺失值(如果有)
df = df.fillna({
    'discount_rate': 0,
    'city': '未知'
})

# 3. 数据类型转换
df['order_date'] = pd.to_datetime(df['order_date'])
df['product_id'] = df['product_id'].astype(int)

# 4. 数据验证
# 检查金额是否为负数
negative_amount = df[df['final_amount'] < 0]

    print(f"\n警告: 发现 {len(negative_amount)} 条负数金额,已删除")
    df = df[df['final_amount'] >= 0]

expensive_products = df[df['price'] > 10000]
if len(expensive_products) > 0:

# 6. 保存清洗后的数据
df.to_csv('ecommerce_orders_cleaned.csv', index=False, encoding='utf-8-sig')
```

```sql
CREATE DATABASE IF NOT EXISTS ecommerce_analysis;

    discount_amount DECIMAL(10, 2),

    INDEX idx_order_date (order_date),
    INDEX idx_city (city),
    INDEX idx_category (category)
);

-- 创建产品表
CREATE TABLE IF NOT EXISTS products (
    product_id INT PRIMARY KEY,

    category VARCHAR(50),

    total_sales INT,

-- 创建用户表
```

#### 6. 编写SQL分析查询(2小时)

**分析任务1: 月度销售趋势**
```sql
-- 月度销售趋势分析
SELECT
    YEAR(order_date) AS year,
    MONTH(order_date) AS month,
    COUNT(*) AS order_count,
    SUM(final_amount) AS total_sales,
    AVG(final_amount) AS avg_order_value,
    SUM(quantity) AS total_items_sold
FROM orders
GROUP BY YEAR(order_date), MONTH(order_date)
ORDER BY year, month;

-- 计算同比增长率
WITH monthly_sales AS (
    SELECT
        YEAR(order_date) AS year,
        MONTH(order_date) AS month,
        SUM(final_amount) AS total_sales
    FROM orders
    GROUP BY YEAR(order_date), MONTH(order_date)
)
SELECT
    year,
    month,
    total_sales,
    LAG(total_sales) OVER(ORDER BY year, month) AS last_month_sales,
    (total_sales - LAG(total_sales) OVER(ORDER BY year, month)) / LAG(total_sales) OVER(ORDER BY year, month) * 100 AS mom_growth,
    LAG(total_sales, 12) OVER(ORDER BY year, month) AS last_year_same_month,
    (total_sales - LAG(total_sales, 12) OVER(ORDER BY year, month)) / LAG(total_sales, 12) OVER(ORDER BY year, month) * 100 AS yoy_growth
FROM monthly_sales
ORDER BY year, month;
```

**分析任务2: 城市销售对比**
```sql
-- 各城市销售分析
SELECT
    city,
    COUNT(*) AS order_count,
    SUM(final_amount) AS total_sales,
    AVG(final_amount) AS avg_order_value,
    SUM(quantity) AS total_items,
    COUNT(DISTINCT user_id) AS unique_customers,
    SUM(final_amount) / COUNT(DISTINCT user_id) AS revenue_per_customer
FROM orders
GROUP BY city
ORDER BY total_sales DESC;
```

**分析任务3: 商品销量TOP10**
```sql
-- 商品销量TOP10
SELECT
    product_id,
    product_name,
    category,
    SUM(quantity) AS total_quantity,
    SUM(final_amount) AS total_revenue,
    COUNT(*) AS order_count,
    AVG(price) AS avg_price,
    RANK() OVER(ORDER BY SUM(final_amount) DESC) AS revenue_rank
FROM orders
GROUP BY product_id, product_name, category
ORDER BY total_revenue DESC
LIMIT 10;

-- 各类别销售占比
SELECT
    category,
    SUM(final_amount) AS total_revenue,
    SUM(final_amount) / (SELECT SUM(final_amount) FROM orders) * 100 AS revenue_percentage,
    COUNT(*) AS order_count,
    COUNT(DISTINCT product_id) AS product_count
FROM orders
GROUP BY category
ORDER BY total_revenue DESC;
```

```sql
    SUM(final_amount) AS total_spent,
    AVG(final_amount) AS avg_order_value,
    MIN(order_date) AS first_order,
    MAX(order_date) AS last_order,
    DATEDIFF(MAX(order_date), MIN(order_date)) AS customer_lifetime_days,
    city
FROM orders
GROUP BY user_id, city
ORDER BY total_spent DESC
LIMIT 100;

-- 用户复购率分析
WITH user_orders AS (
    SELECT
        user_id,
        COUNT(*) AS order_count,
        MIN(order_date) AS first_order,
        MAX(order_date) AS last_order
    FROM orders
    GROUP BY user_id
)
SELECT
    CASE
        WHEN order_count = 1 THEN '首次购买'
        WHEN order_count <= 5 THEN '低频用户(2-5次)'
        WHEN order_count <= 10 THEN '中频用户(6-10次)'
        ELSE '高频用户(>10次)'
    END AS user_segment,
    COUNT(*) AS user_count,
    COUNT(*) / (SELECT COUNT(*) FROM user_orders) * 100 AS user_percentage,
    AVG(order_count) AS avg_orders
FROM user_orders
GROUP BY user_segment
ORDER BY
    CASE user_segment
        WHEN '首次购买' THEN 1
        WHEN '低频用户(2-5次)' THEN 2
        WHEN '中频用户(6-10次)' THEN 3
        ELSE 4
    END;
```
